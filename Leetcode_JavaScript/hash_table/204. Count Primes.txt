/**
 * @param {number} n
 * @return {number}
 */
var countPrimes = function(n) {
    let flag = [], result = 0;
// begin with 2
    for (let i = 2; i < n; i++){
        if (flag[i] === undefined){
            flag[i] = 1;
            result++;
        m = 2;
        while(i * m < n){
// for those which are not prime number, make them "defined". 
            flag[i*m] = 0;
            m++;
            }
        }
    }
    return result;
};
