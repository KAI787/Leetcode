/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var twoSum = function(nums, target) {
    for (let n=0; n<nums.length; n++){
        for (let i=0; i<nums.length; i++){
            if (nums[i]+nums[n]===target && i!=n)
                return [i, n]
        }
    }
        
};
